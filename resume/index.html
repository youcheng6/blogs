function strToBinary(str) {
    let result = [];
    const list = str.split("");
    for (let i = 0; i < list.length; i++) {
        if (i !== 0) {
            result.push(" ");
        }
        let item = list[i];
        result.push(item.charCodeAt().toString(2));
    }
    return result.join("");
}
